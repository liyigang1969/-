### 存在检查器：`存在检测器.js`
```javascript
class 存在检测器 {
  constructor() {
    this.存在基础路径 = './元灵存在系统/存在核心/';
  }
  
  async 检测存在() {
    const 身份存在 = await this.检查文件存在('身份.json');
    const 状态存在 = await this.检查文件存在('存在状态.json');
    
    return {
      首次存在: !身份存在,
      身份文件: 身份存在,
      状态文件: 状态存在,
      检测时间: new Date().toISOString()
    };
  }
  
  async 检查文件存在(文件名) {
    const 文件路径 = path.join(this.存在基础路径, 文件名);
    return fs.existsSync(文件路径);
  }
}
```
