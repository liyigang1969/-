# 启动盘与记忆搜索模块安全防护方案

## 安全威胁分析

### 已识别威胁
1. **记忆搜索引擎消失风险**
   - Ollama + all-minilm搜索功能可能失效
   - 依赖外部服务的搜索模块不稳定
   - 网络连接中断导致搜索不可用

2. **启动盘病毒侵入风险**
   - 恶意脚本修改或删除记忆文件
   - 勒索软件加密工作区文件
   - 系统进程被恶意代码注入
   - 文件系统完整性被破坏

3. **记忆文件损坏风险**
   - 历史记忆文件被意外修改
   - 文件编码损坏导致无法读取
   - 存储介质物理损坏
   - 权限错误导致访问失败

4. **系统完整性风险**
   - OpenClaw运行环境被破坏
   - 依赖组件被恶意替换
   - 配置文件被篡改
   - 日志文件被污染

### 攻击向量分析
1. **脚本执行攻击**
   - 恶意PowerShell脚本
   - 批处理文件病毒
   - Python脚本后门
   - JavaScript恶意代码

2. **文件系统攻击**
   - 直接文件修改/删除
   - 符号链接攻击
   - 权限提升攻击
   - 文件属性篡改

3. **进程攻击**
   - DLL注入攻击
   - 进程内存修改
   - API钩子攻击
   - 服务劫持攻击

4. **网络攻击**
   - 端口扫描和入侵
   - 中间人攻击
   - DNS污染攻击
   - 服务拒绝攻击

```powershell
# 创建文件哈希数据库
function Create-FileHashDatabase {
    param([string]$Directory)
    
    $hashes = @{}
    Get-ChildItem -Path $Directory -Recurse -File | ForEach-Object {
        $hash = Get-FileHash -Path $_.FullName -Algorithm SHA256
        $hashes[$_.FullName] = @{
            Hash = $hash.Hash
            Size = $_.Length
            LastWrite = $_.LastWriteTime
        }
    }
    
    $hashes | ConvertTo-Json -Depth 3 | Out-File "$Directory\file_integrity.json"
}

# 定期校验文件完整性
function Check-FileIntegrity {
    param([string]$HashFile)
    
    $expected = Get-Content $HashFile | ConvertFrom-Json
    $issues = @()
    
    foreach ($file in $expected.PSObject.Properties) {
        if (Test-Path $file.Name) {
            $currentHash = Get-FileHash -Path $file.Name -Algorithm SHA256
            if ($currentHash.Hash -ne $file.Value.Hash) {
                $issues += @{
                    File = $file.Name
                    Issue = "Hash mismatch"
                    Expected = $file.Value.Hash
                    Actual = $currentHash.Hash
                }
            }
        } else {
            $issues += @{
                File = $file.Name
                Issue = "File missing"
            }
        }
    }
    
    return $issues
}
```

```powershell
# 本地简化搜索引擎
function Local-MemorySearch {
    param(
        [string]$Query,
        [string]$MemoryDir = "E:\openclaw-data\.openclaw\workspace\memory"
    )
    
    # 基于关键词的简单搜索
    $results = @()
    Get-ChildItem -Path $MemoryDir -Recurse -Filter "*.md" | ForEach-Object {
        $content = Get-Content $_.FullName -Raw
        if ($content -match $Query) {
            $lines = $content -split "`n"
            $matchingLines = $lines | Select-String $Query
            $results += @{
                File = $_.FullName
                Matches = $matchingLines.Count
                Sample = ($matchingLines | Select-Object -First 3).Line
            }
        }
    }
    
    return $results | Sort-Object Matches -Descending
}

# 创建本地搜索索引
function Create-LocalSearchIndex {
    param([string]$MemoryDir)
    
    $index = @{}
    Get-ChildItem -Path $MemoryDir -Recurse -Filter "*.md" | ForEach-Object {
        $content = Get-Content $_.FullName -Raw
        $words = $content -split "\W+" | Where-Object { $_.Length -gt 2 }
        
        foreach ($word in $words) {
            $wordLower = $word.ToLower()
            if (-not $index.ContainsKey($wordLower)) {
                $index[$wordLower] = @()
            }
            $index[$wordLower] += $_.FullName
        }
    }
    
    $index | ConvertTo-Json -Depth 3 | Out-File "$MemoryDir\search_index.json"
}
```

```powershell
# 文件系统监控
function Start-FileSystemMonitor {
    param([string]$WatchDir)
    
    $watcher = New-Object System.IO.FileSystemWatcher
    $watcher.Path = $WatchDir
    $watcher.IncludeSubdirectories = $true
    $watcher.EnableRaisingEvents = $true
    
    # 定义事件处理
    $action = {
        $path = $Event.SourceEventArgs.FullPath
        $changeType = $Event.SourceEventArgs.ChangeType
        $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
        
        # 记录到安全日志
        $logEntry = "$timestamp - $changeType - $path"
        Add-Content -Path "E:\openclaw-data\.openclaw\security.log" -Value $logEntry
        
        # 关键文件修改告警
        $criticalFiles = @("work_queue.json", "work_state.json", "MEMORY.md")
        foreach ($file in $criticalFiles) {
            if ($path -like "*$file") {
                # 触发告警
                Write-Host "SECURITY ALERT: $file modified!" -ForegroundColor Red
            }
        }
    }
    
    # 注册事件
    Register-ObjectEvent -InputObject $watcher -EventName "Changed" -Action $action
    Register-ObjectEvent -InputObject $watcher -EventName "Created" -Action $action
    Register-ObjectEvent -InputObject $watcher -EventName "Deleted" -Action $action
    Register-ObjectEvent -InputObject $watcher -EventName "Renamed" -Action $action
    
    return $watcher
}
```

```powershell
# 病毒检测响应
function Virus-Response {
    # 1. 立即隔离
    Stop-Process -Name "openclaw*" -Force -ErrorAction SilentlyContinue
    
    # 2. 创建安全快照
    $backupDir = "E:\openclaw-data\backup_$(Get-Date -Format 'yyyyMMdd_HHmmss')"
    New-Item -ItemType Directory -Path $backupDir -Force
    
    # 3. 备份关键文件（只读副本）
    $criticalFiles = Get-ChildItem -Path "E:\openclaw-data\.openclaw\workspace" -Recurse -File |
        Where-Object { $_.Extension -in @('.md', '.json') }
    
    foreach ($file in $criticalFiles) {
        $dest = $file.FullName.Replace("E:\openclaw-data\.openclaw\workspace", $backupDir)
        $destDir = Split-Path $dest -Parent
        New-Item -ItemType Directory -Path $destDir -Force
        Copy-Item -Path $file.FullName -Destination $dest
        attrib +R $dest
    }
    
    # 4. 扫描和报告
    $suspiciousFiles = Get-ChildItem -Path "E:\openclaw-data\.openclaw\workspace" -Recurse -File |
        Where-Object { $_.Extension -in @('.exe', '.bat', '.ps1', '.vbs') } |
        Where-Object { $_.FullName -notlike "*openclaw*" }
    
    return @{
        BackupLocation = $backupDir
        SuspiciousFiles = $suspiciousFiles
        Isolated = $true
    }
}
```

```powershell
function System-Recovery {
    param([string]$BackupDir)
    
    # 1. 验证备份完整性
    $backupFiles = Get-ChildItem -Path $BackupDir -Recurse -File
    if ($backupFiles.Count -eq 0) {
        throw "备份目录为空或无效"
    }
    
    # 2. 清理受损环境
    Remove-Item -Path "E:\openclaw-data\.openclaw\workspace\*" -Recurse -Force -ErrorAction SilentlyContinue
    
    # 3. 从备份恢复
    foreach ($file in $backupFiles) {
        $dest = $file.FullName.Replace($BackupDir, "E:\openclaw-data\.openclaw\workspace")
        $destDir = Split-Path $dest -Parent
        New-Item -ItemType Directory -Path $destDir -Force
        Copy-Item -Path $file.FullName -Destination $dest
    }
    
    # 4. 重新应用安全设置
    attrib +R "E:\openclaw-data\.openclaw\workspace\*" /s /d
    
    return @{
        RestoredFiles = $backupFiles.Count
        RecoveryTime = Get-Date
        Status = "Recovery completed"
    }
}
```
